function uid() { return 'P' + Date.now().toString(36).toUpperCase() + Math.random().toString(36).slice(2,6).toUpperCase(); }


function shuffle(arr) {


const a = [...arr];


for (let i = a.length - 1; i > 0; i--) { const j = Math.floor(Math.random()*(i+1)); [a[i],a[j]]=[a[j],a[i]]; }


return a;


}


function pct(p) { return (p*100).toFixed(0)+'%'; }


function riskColor(p){ return p<=.20?'#22c55e':p<=.40?'#84cc16':p<=.60?'#f59e0b':p<=.80?'#f97316':'#ef4444'; }

function init() {


const p = new URLSearchParams(window.location.search);




// Condition


const c = p.get('condition');


if (c==='none'||c==='low'||c==='high') { S.condition=c; S.conditionSource='url'; }


else { S.condition=Math.random()<.5?'none':'high'; S.conditionSource='random'; }




// Platform detection


const prolificPID = p.get('PROLIFIC_PID') || p.get('prolific_pid');


const mturkWorker = p.get('workerId') || p.get('worker_id');


if (prolificPID) {


S.prolificPID = prolificPID;


S.studyID = p.get('STUDY_ID') || p.get('study_id') || '';


S.sessionID = p.get('SESSION_ID') || p.get('session_id') || '';


S.platform = 'prolific';


S.participantID = prolificPID;


} else if (mturkWorker) {


S.platform = 'mturk';


S.participantID = mturkWorker;


} else {


S.platform = 'manual';


}




// Participant PID fallback


S.pid = p.get('pid') || S.participantID || uid();


S.trialOrder = shuffle([...CFG.probabilities]);


render();


}

function checkComp(q1c, q2c) {


const a1=S.compAnswers.q1, a2=S.compAnswers.q2;


if (!a1||!a2) { showErr('comp-error','Please answer both questions.'); return; }


['a','b','c'].forEach(v=>{


const e1=document.getElementById('q1-'+v), e2=document.getElementById('q2-'+v);


[e1,e2].forEach(e=>e.classList.remove('is-correct','is-wrong'));


if(v===q1c) e1.classList.add('is-correct'); else if(v===a1&&a1!==q1c) e1.classList.add('is-wrong');


if(v===q2c) e2.classList.add('is-correct'); else if(v===a2&&a2!==q2c) e2.classList.add('is-wrong');


});


if(a1!==q1c||a2!==q2c){


S.compErrors++;


showErr('comp-error','Some answers incorrect (red). Correct answers highlighted green. Please re-read and try again.');


return;


}


S.compAnswers={q1:null,q2:null};


go('practice');


}




function showErr(id, msg) {

if(el){el.textContent=msg;el.classList.remove('hidden');}


}

function startCountdown() {

let remaining=S.countdownRemaining;


const waitStart=Date.now();


S.countdownInterval=setInterval(()=>{


remaining--;


const num=document.getElementById('countdown-num');


const ring=document.getElementById('ring-fill');


const lbl=document.getElementById('countdown-label');


if(!num){clearInterval(S.countdownInterval);return;}


num.textContent=remaining;


if(ring) ring.style.strokeDashoffset=(circ*(1-remaining/RESEARCHER.waitSecs)).toFixed(1);


if(remaining<=10&&lbl) lbl.textContent='Almost there…';


if(remaining<=0){


clearInterval(S.countdownInterval); S.countdownInterval=null;


const resp=S.responses[S.responses.length-1];


if(resp) resp.rt_wait_ms=Date.now()-waitStart;


if(num) num.textContent='0';


if(lbl) lbl.textContent='Done!';


const done=document.getElementById('countdown-done');


if(done) done.classList.remove('hidden');

// GOOGLE SHEETS AUTO-SAVE


// (Called automatically when end screen loads)


// ═══════════════════════════════════════════════════════════════


function attemptSheetsSave() {


if (!RESEARCHER.sheetsWebAppURL) {


updateSaveUI('failed', 'Auto-save not configured. Please use the Download buttons below to save your data.');


return;


}


updateSaveUI('saving', 'Saving to researcher\'s server…');


const payload = buildPayload();


// Use GET with URL-encoded JSON (avoids CORS preflight)


const url = RESEARCHER.sheetsWebAppURL + '?data=' + encodeURIComponent(JSON.stringify(payload));


fetch(url, { mode: 'no-cors' })


.then(() => updateSaveUI('saved', '✅ Data saved to researcher\'s server. Please also download CSV as backup.'))


.catch(() => updateSaveUI('failed', '⚠️ Auto-save failed (network error). Please download your data below.'));


}




function updateSaveUI(status, msg) {


S.saveStatus = status;


const dot = document.getElementById('save-dot');


const txt = document.getElementById('save-msg');


if (dot) { dot.className = 'save-dot ' + status; }


if (txt) { txt.textContent = msg; }


}

// DATA EXPORT (CSV + JSON)

function dlCSV() {


const cols = [


'pid','prolific_pid','participant_name','participant_email',


'platform','condition','condition_label','condition_source',


'trial','probability','seek_now',


'reassurance','fear','action',


'rt_decision_ms','rt_wait_ms',


'comp_errors','timestamp'


];


const ts = new Date().toISOString();


const rows = S.responses.map(r => [


S.pid, S.prolificPID||'', S.participantName, S.participantEmail,


S.platform, S.condition, cond().label, S.conditionSource,


r.trial, r.probability.toFixed(2), r.seek_now,


r.reassurance??'', r.fear??'', r.action??'',


r.rt_decision_ms, r.rt_wait_ms??0,


S.compErrors, ts


].map(v=>String(v).includes(',') ? `"${v}"` : v).join(','));


dlFile([cols.join(','),...rows].join('\n'), `RS_${S.pid}.csv`, 'text/csv');


}




function dlJSON() {


dlFile(JSON.stringify(buildPayload(), null, 2), `RS_${S.pid}.json`, 'application/json');


}




function dlFile(content, name, mime) {


const url=URL.createObjectURL(new Blob([content],{type:mime}));


const a=Object.assign(document.createElement('a'),{href:url,download:name});


document.body.appendChild(a); a.click(); document.body.removeChild(a);


URL.revokeObjectURL(url);


}
